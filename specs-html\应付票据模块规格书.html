1. 模块概述

1.1 业务定位

应付票据模块是ERP财务管理中采购端票据支付的核心枢纽，负责管理我方开出的商业汇票从开立、交付、持有、到期兑付的全生命周期。

CREATE TABLE payable_bill (
    id              BIGINT PRIMARY KEY AUTO_INCREMENT,
    bill_no         VARCHAR(20)  NOT NULL UNIQUE COMMENT '应付票据编号 PBILL+年月+4位序号',

    due_date        DATE         NOT NULL COMMENT '到期日期',
    face_amount     DECIMAL(18,2) NOT NULL COMMENT '票面金额',
    
    -- 票据当事人

    drawer_tax_no   VARCHAR(20)  DEFAULT NULL COMMENT '出票人税号（我方税号）',
    payee           VARCHAR(100) NOT NULL COMMENT '收款人名称（供应商）',
    payee_id        BIGINT       DEFAULT NULL COMMENT '收款人供应商ID',
    payee_bank      VARCHAR(100) DEFAULT NULL COMMENT '收款人开户行',
    payee_acct      VARCHAR(30)  DEFAULT NULL COMMENT '收款人账号',
    acceptor        VARCHAR(100) NOT NULL COMMENT '承兑人名称（银承=银行/商承=我方）',
    acceptor_type   TINYINT      NOT NULL COMMENT '承兑人类型: 1=银行 2=企业（我方）',
    
    -- 关联信息
    payment_id      BIGINT       DEFAULT NULL COMMENT '关联付款单ID',
    ap_detail_id    BIGINT       DEFAULT NULL COMMENT '关联应付明细ID',

    margin_amount   DECIMAL(18,2) DEFAULT NULL COMMENT '保证金金额（银承专用）',

4. 业务规则定义

4.1 票据编号规则 

格式: PBILL + YYYYMM + 4位序号
示例: PBILL2026040001

规则:
- 按月归零重新计数
- 同一组织内唯一
- 一旦生成不可修改
- 作废票据编号保留（不重用）
- 与应收票据编号(BILL)区分，便于识别

4.2 应付票据状态枚举与状态机

4.3 开票金额校验规则 

def validate_payable_bill(bill: dict) -> list[str]:
    """应付票据开立校验"""
    errors = []
    
    # 必填字段
    required = ['bill_type', 'draft_date', 'due_date', 'face_amount', 
                'payee', 'payee_id', 'acceptor']
    for field in required:
        if not bill.get(field):
            errors.append(f"缺少必填字段: {field}")
    
    # 业务规则
    if bill.get('face_amount', 0) <= 0:
        errors.append("票面金额必须大于0")
    
    if bill.get('ap_amount', 0) > bill.get('face_amount', 0):
        errors.append("抵付金额不能超过票面金额")
    
    if bill.get('due_date') and bill.get('draft_date'):
        if bill['due_date'] <= bill['draft_date']:
            errors.append("到期日期必须晚于出票日期")
        days = (bill['due_date'] - bill['draft_date']).days
        if days > 365:
            errors.append("票据期限不得超过一年")
    
    # 银承必须填写保证金信息
    if bill.get('bill_type') == 1:
        if not bill.get('margin_ratio'):
            errors.append("银行承兑汇票必须填写保证金比例")
    
    # 商承限制（信用评估）
    if bill.get('bill_type') == 2:
        if not has_credit_check(bill['drawer_tax_no']):
            errors.append("商业承兑汇票需要通过企业信用评估")
    
    return errors

4.4 审批金额阈值 

def get_approval_chain(amount: float) -> list[str]:
    """根据票据金额返回审批链"""
    if amount < 100000:        # < 10万
        return ['cashier', 'finance_supervisor']
    elif amount < 500000:      # 10-50万
        return ['cashier', 'finance_supervisor', 'finance_manager']
    else:                      # ≥ 50万
        return ['cashier', 'finance_supervisor', 'finance_manager', 'general_manager']

4.5 背书收票验证规则 

def validate_received_bill_for_payment(bill: dict, ap_detail: dict) -> list[str]:
    """背书收票抵付验证"""
    errors = []
    
    # 票面金额校验
    if bill['face_amount'] > ap_detail['remaining_amount']:
        errors.append(f"票据金额({bill['face_amount']})超过应付余额({ap_detail['remaining_amount']})")
    
    # 到期日合理性
    days_to_due = (bill['due_date'] - date.today()).days
    if days_to_due < 7:
        errors.append(f"票据距到期仅{days_to_due}天，时间过短不建议接受")
    if days_to_due < 0:
        errors.append("票据已过期，不能接受")
    
    # 商业承兑汇票信用检查
    if bill['bill_type'] == 2:
        credit_rating = get_credit_rating(bill['acceptor_tax_no'])
        if credit_rating and credit_rating < 'B':
            errors.append(f"承兑人信用评级{credit_rating}过低，建议拒绝或要求补担保")
    
    # 背书连续性
    if bill.get('is_paper', False):
        if not check_endorsement_chain(bill):
            errors.append("背书链不连续，请核实")
    
    return errors

7. 备付金管理机制

7.1 备付金预测规则 

def calculate_reserve_forecast(days: int = 30) -> dict:
    """
    计算未来N天的应付票据备付金需求
    """
    today = date.today()
    end_date = today + timedelta(days=days)
    
    # 查询持有中的应付票据
    bills = PayableBill.query.filter(
        PayableBill.status == PayableBillStatus.HOLDING,
        PayableBill.due_date.between(today, end_date),
        PayableBill.is_deleted == 0
    ).order_by(PayableBill.due_date).all()
    
    # 银承 vs 商承分类
    bank_acceptance = [b for b in bills if b.bill_type == 1]
    commercial_acceptance = [b for b in bills if b.bill_type == 2]
    
    return {
        'total': sum(b.face_amount for b in bills),
        'bank_acceptance_total': sum(b.face_amount for b in bank_acceptance),
        'commercial_acceptance_total': sum(b.face_amount for b in commercial_acceptance),
        'bills': bills,
        'margin_already_deposited': sum(b.margin_amount for b in bank_acceptance),
        'actual_cash_needed': sum(b.face_amount for b in commercial_acceptance) + 
            sum(b.face_amount - b.margin_amount for b in bank_acceptance)
    }

7.2 备付金预警

级别条件措施
🟢 正常可用资金 ≥ 应付票据到期总额无需额外操作
🟡 关注可用资金 ≥ 总额的80%通知资金岗关注
🟠 警告可用资金 ≥ 总额的50%启动头寸调拨流程
🔴 紧急可用资金 < 总额的50%紧急资金调度+管理层预警

7.3 银承保证金管理 

class MarginRule:
    """银行承兑汇票保证金规则"""
    
    # 保证金比例参考（具体以银行协议为准）
    MARGIN_RATIOS = {
        'AAA': 0,        # AAA级企业免保证金
        'AA': 10,        # AA级10%
        'A': 30,         # A级30%
        'BBB': 50,       # BBB级50%
        'BB_below': 100  # BB及以下全额保证金
    }
    
    @staticmethod
    def calculate_margin(face_amount: float, credit_rating: str) -> dict:
        ratio = MarginRule.MARGIN_RATIOS.get(credit_rating, 100)
        margin = round(face_amount * ratio / 100, 2)
        return {
            'face_amount': face_amount,
            'credit_rating': credit_rating,
            'margin_ratio': ratio,
            'margin_amount': margin,
            'free_amount': face_amount - margin
        }

8. 安全与风控

8.1 应付票据风险控制

风控点规则
开票总额控制单一供应商应付票据总额不超过应付余额的80%
商承信用要求开出商业承兑汇票前必须通过企业信用评估
大额审批票面金额≥50万需总经理审批
到期违约预警距到期3天资金未到位，升级为紧急事件
展期限制同一票据最多展期2次，总期限不超过12个月
退回监控月度退回率超过10%触发内部审查

8.2 操作权限

操作角色说明
票据开立申请出纳/财务基础权限
开票审批财务主管/财务总监/总经理按金额阈值
纸质票交付出纳实物交接
到期付款确认出纳（执行）+ 财务主管（确认）双人控制
票据展期财务主管（申请）+ 财务总监（审批）影响付款节奏
背书收票验证出纳（接收）+ 财务主管（审批）防止假票/瑕疵票
台账查询出纳/财务/主管按角色过滤
备付金预测资金岗/财务主管资金规划

8.3 银企对接安全

措施说明
付款指令校验付款金额+收款方+票据号三重核对
U盾双签大额付款（≥10万）需U盾双签
指令追踪每笔银企指令全链路可查
异常拦截超时/失败指令自动冻结，人工处理
